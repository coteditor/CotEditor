\begin{theorem}
For all integers $n \ge 1$, we have
\[
  \sum_{k=1}^{n} k = \frac{n(n+1)}{2}.
\]
\end{theorem}

\begin{proof}
The statement follows by induction on $n$.
\end{proof}
